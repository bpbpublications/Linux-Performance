pstree_cpu() {

  local indent=$2

      pstree_cpu $child "$indent  "
    done
  fi
}

find_high_cpu() {
  threshold=${1:-50}  # Default threshold of 50%
  for pid in $(pgrep -d' ' ".*"); do
    cpu=$(ps -p $pid -o %cpu --no-headers | tr -d ' ')
    if (( $(echo "$cpu > $threshold" | bc -l) )); then

  done

#!/bin/bash
# Disk I/O Utilization Monitor (Auto-detects disks)
# Logs if any disk is over 90% utilization


LOGFILE="/var/log/disk_alert.log"
THRESHOLD=90


# Detect all disks (ignores loop devices, RAM disks, partitions)
DISKS=$(lsblk -ndo NAME,TYPE | awk '$2=="disk"{print $1}')


while true; do
    for DISK in $DISKS; do
        UTIL=$(iostat -x "$DISK" 1 2 | tail -n 1 | awk '{print $NF}')
        
        # If UTIL is numeric and above threshold
        if [[ $UTIL =~ ^[0-9]+(\.[0-9]+)?$ ]]; then
            if awk -v util="$UTIL" -v thr="$THRESHOLD" 'BEGIN {exit !(util > thr)}'; then
                {
                    echo "[$(date)] High disk utilization detected on /dev/$DISK: ${UTIL}%"
                    iostat -x "$DISK" 1 3
                    iotop -b -n 1 -o
                    echo "----------------------------------------"
                } >> "$LOGFILE"
            fi
        fi
    done
    sleep 10
done

# Monitor CPU utilization using /proc/stat
function cpu_usage() {
  local prev_total=0 prev_idle=0
  while true; do
    local cpu=($(head -n1 /proc/stat))
    local user=${cpu[1]} nice=${cpu[2]} system=${cpu[3]} idle=${cpu[4]} iowait=${cpu[5]} irq=${cpu[6]} softirq=${cpu[7]}
    local total=$((user + nice + system + idle + iowait + irq + softirq))
    local diff_total=$((total - prev_total))
    local diff_idle=$((idle - prev_idle))
    local utilization=$(( 100 * (diff_total - diff_idle) / diff_total ))
    echo "CPU utilization: $utilization%"
    prev_total=$total
    prev_idle=$idle
    sleep 1
  done
}

function system_monitor() {
  # Memory utilization
  local mem=($(grep -E 'MemTotal|MemFree|Buffers|Cached' /proc/meminfo | awk '{print $2}'))
  local mem_total=${mem[0]} mem_free=${mem[1]} buffers=${mem[2]} cached=${mem[3]}
  local mem_used=$((mem_total - mem_free - buffers - cached))
  local mem_percent=$((mem_used * 100 / mem_total))
  
  # CPU load
  local load=$(cut -d ' ' -f 1-3 /proc/loadavg)
  
  # Disk utilization
  local disk_stats=$(grep 'sda ' /proc/diskstats)
  
  echo "Memory: $mem_percent% | Load: $load | Disk: $disk_stats"
}

BPF_HASH(start, struct request *);

void trace_start(struct pt_regs *ctx, struct request *req) {
    // Record start time
    u64 ts = bpf_ktime_get_ns();

void trace_completion(struct pt_regs *ctx, struct request *req) {

}

from bcc import BPF
from time import sleep


# BPF program
bpf_text = """
#include <uapi/linux/ptrace.h>


struct alloc_info_t {
    u64 size;    // Requested allocation size
    u64 timestamp;  // Timestamp of allocation
};


BPF_HASH(allocs, u64, struct alloc_info_t);
BPF_HISTOGRAM(size_hist);


int trace_malloc(struct pt_regs *ctx) {
    u64 size = PT_REGS_PARM1(ctx);
    u64 timestamp = bpf_ktime_get_ns();
    u64 addr = PT_REGS_RC(ctx);
    
    if (addr != 0) {
        struct alloc_info_t info = {.size = size, .timestamp = timestamp};
        allocs.update(&addr, &info);
        
        // Update size histogram
        size_hist.increment(bpf_log2l(size));
    }
    
    return 0;
}


int trace_free(struct pt_regs *ctx) {
    u64 addr = PT_REGS_PARM1(ctx);
    struct alloc_info_t *info = allocs.lookup(&addr);
    
    if (info != 0) {
        u64 current = bpf_ktime_get_ns();
        u64 lifetime = (current - info->timestamp) / 1000000; // ms
        
        bpf_trace_printk("freed %lu bytes allocated at %lx after %lu ms\\n", 
                         info->size, addr, lifetime);
        allocs.delete(&addr);
    }
    
    return 0;
}
"""


b = BPF(text=bpf_text)


# Attach to libc malloc and free
b.attach_uprobe(name="c", sym="malloc", fn_name="trace_malloc", retprobe=1)
b.attach_uprobe(name="c", sym="free", fn_name="trace_free")


print("Tracing malloc() and free() calls... Hit Ctrl-C to end.")


# Print output
try:
    while True:
        print(b.trace_fields())

# BCC program to correlate syscalls with block I/O
from bcc import BPF


bpf_text = """
#include <uapi/linux/ptrace.h>
#include <linux/blkdev.h>


struct event_t {
    u32 pid;
    char comm[16];
    char syscall[16];
    char operation[8];
    u64 latency;
};


BPF_HASH(start, u32);
BPF_PERF_OUTPUT(events);


int trace_read_entry(struct pt_regs *ctx) {
    u64 ts = bpf_ktime_get_ns();
    u32 pid = bpf_get_current_pid_tgid();
    start.update(&pid, &ts);
    return 0;
}


int trace_block_io(struct pt_regs *ctx) {
    u32 pid = bpf_get_current_pid_tgid();
    u64 *tsp = start.lookup(&pid);
    
    if (tsp != 0) {
        struct event_t event = {};
        event.pid = pid;
        bpf_get_current_comm(&event.comm, sizeof(event.comm));
        
        u64 delta = bpf_ktime_get_ns() - *tsp;
        event.latency = delta / 1000; // microseconds
        
        __builtin_memcpy(&event.syscall, "read", 5);
        __builtin_memcpy(&event.operation, "disk", 5);
        
        events.perf_submit(ctx, &event, sizeof(event));
    }
    
    return 0;
}
"""

        comm_match = comm_pattern.search(line)
        pid_match = pid_pattern.search(line)
        
        if syscall_match and comm_match and pid_match:
            syscall = syscall_match.group(1)
            comm = comm_match.group(1)
            pid = int(pid_match.group(1))
            
            key = (syscall, comm, pid)
            syscall_counts[key] = syscall_counts.get(key, 0) + 1
            
        # Flush counts to database every 60 seconds
        now = time.time()
        if now - last_flush >= 60:
            timestamp = int(now)
            
            for (syscall, comm, pid), count in syscall_counts.items():
                c.execute(
                    'INSERT INTO syscall_metrics VALUES (?, ?, ?, ?, ?)',
                    (timestamp, syscall, comm, pid, count)
                )
            
            conn.commit()
            syscall_counts.clear()
            last_flush = now

import multiprocessing
import time
import random
import os


def cpu_load(duration, intensity):
    """Generate CPU load at specified intensity (0-1)"""
    end_time = time.time() + duration
    while time.time() < end_time:
        # Adjust work/sleep ratio to match intensity
        start_work = time.time()
        # Compute-intensive operation
        _ = [i**2 for i in range(10000)]
        work_time = time.time() - start_work
        # Sleep to achieve desired intensity
        if work_time < 0.01:  # Prevent division by zero
            time.sleep(0.01 * (1 - intensity) / intensity)
        else:
            sleep_time = work_time * (1 - intensity) / intensity
            if sleep_time > 0:
                time.sleep(sleep_time)


def memory_load(duration, memory_mb):
    """Allocate and use specified memory amount"""
    # Allocate memory
    data = bytearray(memory_mb * 1024 * 1024)
    end_time = time.time() + duration
    while time.time() < end_time:
        # Randomly access and modify memory to prevent optimization
        for _ in range(1000):
            idx = random.randrange(0, len(data) - 1)
            data[idx] = random.randint(0, 255)
        time.sleep(0.1)

import os
import time
import random


def io_pattern_generator(directory, file_count, file_size_mb, 
                        read_pct=50, sequential_pct=50, duration=60):
    """
    Generate custom I/O patterns with controlled read/write ratio
    and sequential/random access patterns
    """
    # Create test files if they don't exist
    files = []
    for i in range(file_count):
        filename = os.path.join(directory, f"test_file_{i}.dat")
        if not os.path.exists(filename):
            with open(filename, 'wb') as f:
                f.write(os.urandom(file_size_mb * 1024 * 1024))
        files.append(filename)
    
    end_time = time.time() + duration
    operations = 0
    
    while time.time() < end_time:
        # Select file
        filename = random.choice(files)
        file_size = os.path.getsize(filename)
        
        # Determine operation type (read or write)
        is_read = random.randrange(100) < read_pct
        
        # Determine access pattern (sequential or random)
        is_sequential = random.randrange(100) < sequential_pct
        
        with open(filename, 'rb+') as f:
            if is_sequential:
                # Sequential access - start at beginning
                position = 0
                f.seek(position)
            else:
                # Random access - choose random position
                position = random.randrange(0, file_size - 1024)
                f.seek(position)
            
            if is_read:
                # Read operation
                data = f.read(1024)
            else:
                # Write operation
                f.write(os.urandom(1024))
        
        operations += 1
    
    return operations

import yaml
import time
import subprocess
import multiprocessing


def run_workload_scenario(scenario_file):
    """
    Run a multi-phase workload scenario defined in YAML
    
    Example YAML structure:
    name: "Database server load test"
    phases:
      - name: "Warmup"
        duration: 60
        components:
          - type: "cpu"
            intensity: 0.3
            cores: 2
          - type: "memory"
            size_gb: 1
      - name: "Peak load"
        duration: 300
        components:
          - type: "cpu"
            intensity: 0.8
            cores: 4
          - type: "memory"
            size_gb: 4
          - type: "io"
            read_pct: 70
            file_count: 20
    """
    with open(scenario_file, 'r') as f:
        scenario = yaml.safe_load(f)
    
    print(f"Running scenario: {scenario['name']}")
    
    for phase in scenario['phases']:
        print(f"Starting phase: {phase['name']} (duration: {phase['duration']}s)")
        processes = []
        
        # Launch each component
        for component in phase['components']:
            if component['type'] == 'cpu':
                cores = component.get('cores', 1)
                intensity = component.get('intensity', 0.5)
                for _ in range(cores):
                    p = multiprocessing.Process(
                        target=cpu_load, 
                        args=(phase['duration'], intensity)
                    )
                    p.start()
                    processes.append(p)
            
            elif component['type'] == 'memory':
                size_gb = component.get('size_gb', 1)
                p = multiprocessing.Process(
                    target=memory_load, 
                    args=(phase['duration'], size_gb * 1024)
                )
                p.start()
                processes.append(p)
            
            elif component['type'] == 'io':
                p = multiprocessing.Process(
                    target=io_pattern_generator,
                    args=(
                        "/tmp/test",
                        component.get('file_count', 10),
                        component.get('file_size_mb', 100),
                        component.get('read_pct', 50),
                        component.get('sequential_pct', 50),
                        phase['duration']
                    )
                )
                p.start()
                processes.append(p)
        
        # Wait for phase to complete
        for p in processes:
            p.join()
        
        print(f"Completed phase: {phase['name']}")
    
    print(f"Scenario completed: {scenario['name']}")
